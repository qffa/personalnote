# 节点操作

## 节点

**节点概念**

网页中所有内容都是节点，在DOM中节点用node表示。常见节点：

- 标签
- 属性
- 文本
- 注释
...

HTML DOM树中所有节点都可以通过JavaScript访问，所有节点均可以被修改，也可以创建或删除

**节点属性**

节点至少拥有三个基本属性

- nodeType
- nodeName
- nodeValue

**节点类型**

- 元素节点，nodeType 为 1
- 属性节点，nodeType 为 2
- 文本节点，nodeType 为 3

*文本节点包含文字、空格和换行等*

*实际开发中，主要是操作元素节点*

**节点层级**

父子兄

- 父节点

```js
element.parentNode
```

- 子节点

```js
element.childNodes
element.children (非标准，但得到了所有浏览器的支持)

element.firstChild
element.lastChild

// IE9以上支持
element.firstElementChild
element.lastElementChild

// 无兼容性问题
element.children[0]
element.children[element.children.length-1]
```

- 兄弟节点

```js
element.nextSibling
element.previousSibling

// IE9以上支持
element.nextElementSibling
element.previousElementSibling
```

## 获取元素

**利用DOM提供的方法**

```js
document.getElementById()
document.getElementsByTagName()
document.querySelector()
...
```

- 逻辑性不强
- 繁琐

**利用节点层级关系获取**

利用节点的 **父子兄** 关系获取元素

- 逻辑性强
- 但兼容性稍差

**例子**

```
<body>
  <div class="box">
    <div class="qr"></div>
  </div>
  <ul>
    <li>1</li>
    <li>2</li>
    <li>3</li>
  </ul>
  <script>
    qr = document.querySelector('.qr');

    // 获取父节点
    console.log(qr.parentNode);

    ul = document.querySelector('ul');

    // 获取所有子节点
    console.log(ul.childNodes);
    console.log(ul.childNodes[0].nodeType);

    // 获取所有元素子节点
    console.log(ul.children);

    // 获取第一个或最后一个节点
    console.log(ul.firstChild);
    console.log(ul.lastChild);

    // 获取第一个或最后一个元素节点，IE9以上支持
    console.log(ul.firstElementChild);
    console.log(ul.lastElementChild);

    // 获取第一个或最后一个元素节点，无兼容性问题
    console.log(ul.children[0]);
    console.log(ul.children[ul.children.length-1]);
  </script>
</body
```

## 创建节点


**创建节点**

```js
document.createElement('tagName')
```

**添加节点**

```js
//追加子元素
node.appendChild(child)
//在指定子元素前添加元素
node.insertBefore(child, 指定元素)
```

**例子**

```html
<body>
  <ul>
    <li>121</li>
  </ul>
  <script>
    let li = document.createElement('li');
    let ul = document.querySelector('ul');
    let li2 = document.createElement('li');
    ul.appendChild(li);
    ul.insertBefore(li2, ul.children[0]);
  </script>
</body>
```

**案例**

简易留言板

```html
<body>
  <textarea name="" id="" cols="30" rows="10"></textarea>
  <button>发布</button>
  <ul></ul>
  <script>
    let btn = document.querySelector('button');
    let text = document.querySelector('textarea');
    let ul = document.querySelector('ul');

    btn.onclick = function() {
      if (text.value === '') {
        alert('您没有输入内容');
        return false
      }
      else {
        let li = document.createElement('li');
        li.innerHTML = text.value;
        // ul.appendChild(li);
        ul.insertBefore(li, ul.children[0]);
      }
    }
  </script>
</body>
```

## 删除节点

```js
node.removeChild(child)
```

**例子**

```html
<body>
  <button>删除</button>
  <ul>
    <li>熊大</li>
    <li>熊二</li>
    <li>光头强</li>

  </ul>
  <script>
    let ul = document.querySelector('ul');
    let btn = document.querySelector('button');

    btn.onclick = function () {
      if (ul.children.length === 0) {
        this.disabled = true;
      } else {
        ul.removeChild(ul.children[0]);
      }
    }
  </script>
</body>
```


**案例**

留言板删除功能

```html
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    ul {
      width: 200px;
    }

    li {
      line-height: 1.6;
    }

    li a {
      float: right;
    }
  </style>
</head>

<body>
  <textarea name="" id="" cols="30" rows="10"></textarea>
  <button>发布</button>
  <ul></ul>
  <script>
    let btn = document.querySelector('button');
    let text = document.querySelector('textarea');
    let ul = document.querySelector('ul');

    btn.onclick = function() {
      if (text.value === '') {
        alert('您没有输入内容');
        return false
      }
      else {
        let li = document.createElement('li');
        li.innerHTML = text.value + "<a href='javascript:void(0)'>删除</a>";
        // ul.appendChild(li);
        ul.insertBefore(li, ul.children[0]);

        // 注册删除事件
        let as = document.querySelectorAll('a');
        for (let i = 0; i < as.length; i++) {
          as[i].onclick = function() {
            ul.removeChild(this.parentNode);
          }
        }
      }
    }
  </script>
</body>
```

## 复制节点

```js
node.cloneNode()
```

参数
- `false`: 浅拷贝，只复制标签，不复制里面的内容和子元素（默认值）
- `true`: 深拷贝，拷贝标签内所有内容

**例子**

```html
<body>
  <ul>
    <li>111</li>
    <li>2</li>
    <li>3</li>
  </ul>

  <script>
    let ul = document.querySelector('ul');
    let newli = ul.children[0].cloneNode(true);

    ul.appendChild(newli);
  </script>
</body>
```


## 综合案例

动态生成表格


```html
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    * {
      margin: 0;
      padding: 0;
    }

    body {
      font-family: Microsoft Yahei;
    }

    table {
      width: 500px;
      margin: 20px auto;
      line-height: 2;
      border-collapse: collapse;
      border-spacing: 0;
      text-align: center;
    }

    thead > tr {
      border-bottom: 2px solid #d7d7d7;
    }

    tbody > tr {
      border-bottom: 1px solid #d7d7d7;
    }


  </style>
</head>
<body>
  <table>
    <thead>
      <tr>
        <th>姓名</th>
        <th>科目</th>
        <th>成绩</th>
        <th>操作</th>
      </tr>
    </thead>
    <tbody>
    </tbody>
  </table>

  <script>
    let datas = [
      {
        name: 'Momo',
        subject: 'Javascript',
        score: 99
      },
      {
        name: 'Molang',
        subject: 'Javascript',
        score: 90
      },
      {
        name: 'Piupiu',
        subject: 'Javascript',
        score: 80
      },
      {
        name: 'Keke',
        subject: 'Javascript',
        score: 95
      }
    ]

    let tbody = document.querySelector('tbody');

    // 遍历行
    for (let i = 0; i < datas.length; i++) {
      let tr = document.createElement('tr');

      tbody.appendChild(tr);

      // 遍历列
      for (let k in datas[i]) {
        let td = document.createElement('td');
        td.innerHTML = datas[i][k];
        tr.appendChild(td);
      }

      // 删除按钮
      let td = document.createElement('dt');
      td.innerHTML = '<a href="javascript:;">删除</a>';
      tr.appendChild(td);
    }

    // 删除功能
    var as = document.querySelectorAll('a');
    for (let i = 0; i < as.length; i++) {
      as[i].onclick = function() {
        tbody.removeChild(this.parentNode.parentNode);
      }
    }
  </script>
</body>
```
