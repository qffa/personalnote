# JSON


**json.Marshal**

编码：其他格式转JSON

格式：

```go
func Marshal(v interface()) {[]byte, error}
```

例子：

```go
package main

import (
    "log"
    "fmt"
    "encoding/json"
)

type foo struct {
  Message string
  Age int
  Name string
  surname string  // 变量名小写，转不到json里面去哦
}

func main(){
  data, err = json.Marshal(&foo("4SCore", 56, "Abe", "Lincoln"))
  fmt.Print(string(data))
  if err != nil {
    log.Fatal(err)
  }
}
```

**格式化json字符串**

带缩进的格式

```go
json.MarshalIndent(v, "", "  ")
```


**json.Unmarshal**

解码：JSON转其他格式

格式：

```go
func Unmarshal(data []byte, v interface{}) error
```

例子：

```go
package main

import (
    "log"
    "fmt"
    "encoding/json"
)

type foo struct {
    Message string
    Age int
    Name string
    surname string
}

func main() {
    //data, _ := json.Marshal(&foo{"4Score", 56, "Abe", "Lincoln"})
    //fmt.Print(string(data))

    f := foo{}
    err := json.Unmarshal([]byte(`{"Message":"4SCore", "Age":56, "Name":"Abe"}`), &f)
    if err != nil {
          log.Fatal(err)
    }

    fmt.Println(f.Message)  //4SCore

}

```

**结构体tag**

```go
type foo struct {
  // 需要加双引号，而且冒号后面没有空格
  Message string    `json:"message,omitempty"`
  Age int           `json:"age,omitempty"`
  Name string       `json:"firstname,omitempty"`
  surname string  // 变量名小写，转不到json里面去哦
}
```

指定json编码后的名称

使用反射实现


**测试代码**

```go
package main

import (
    "fmt"
    "log"
    "encoding/json"
)


type Product struct {
    ProductID       int     `json:"productID"`
    Manufacturer    string  `json:"manufacturer"`
    QuantityOnHand  int     `json:"quantityOnHand"`
    ProductName     string  `json:"productName"`
}


func main(){
    product := &Product{
        ProductID:      123,
        Manufacturer:   "Big Box Company",
        QuantityOnHand: 28,
        ProductName:    "中文",
    }

    productJSON,err := json.Marshal(product)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(string(productJSON))

    productJSON = []byte(`{
        "productID": 456,
        "manufacturer": "Small Box Company",
        "sku": "8ehfhsdiw",
        "upc": "037466366",
        "pricePerUnit": "$9.9",
        "quantityOnHand": 18,
        "productName": "Sprocketdd"
        }`)


    product = &Product{}

    err = json.Unmarshal([]byte(productJSON), product)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(product)
}
```
