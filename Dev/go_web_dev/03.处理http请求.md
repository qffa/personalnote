# 处理http请求

## Request 对象

- 请求方法
- 请求头
- 请求体


**Request.Method**

- string

**Request.Header**

- Header(map[string][]strinig)

**Request.Body**

- io.ReadCloser
- return EOF when not present



## 示例


GET + POST

```go
package main

import (
    "fmt"
    "log"
    "encoding/json"
    "net/http"
    "io/ioutil"
)

type Product struct {
    ProductID       int     `json: productId`
    Manufacturer    string  `json: manufacturer`
    Sku             string  `json: sku`
    Upc             string  `json: upc`
    PricePerUnit    string  `json: pricePerUnit`
    QuantityOnHand  int     `json: quantityOnHand`
    ProductName     string  `json: productName`
}

var productList []Product

// 把数据先存内存里面
func init(){
    productsJSON := `[
        {
                        "productId": 1,
                        "manufacturer": "jenkins",
                        "sku": "kdieihw8hy7",
                        "upc": "883772837827",
                        "pricePerUnit": "497.56",
                        "quantityOnHand": 9876,
                        "productName": "skiky node"
                },
                {
                        "productId": 2,
                        "manufacturer": "Docker",
                        "sku": "k8eu2da4g",
                        "upc": "836466559",
                        "pricePerUnit": "496.12",
                        "quantityOnHand": 63649,
                        "productName": "docker hub"
                },
                {
                        "productId": 3,
                        "manufacturer": "google",
                        "sku": "87hdhdy3g",
                        "upc": "947563883385757",
                        "pricePerUnit": "4123.98",
                        "quantityOnHand": 4857,
                        "productName": "kubernetes"
                }
    ]`

        err := json.Unmarshal([]byte(productsJSON), &productList)
    if err != nil {
        fmt.Println("error json")
        log.Fatal(err)
    }
}

func getNextID() int {
    higestID := -1
    for _, product := range productList {
        if higestID < product.ProductID {
            higestID = product.ProductID
        }
    }
    return higestID + 1
}


func productsHandler(w http.ResponseWriter, r *http.Request) {
    switch r.Method {
    case http.MethodGet:
        productsJSON, err := json.Marshal(productList)
        if err != nil {
            w.WriteHeader(http.StatusInternalServerError)
            return
        }
        w.Header().Set("Content-Type", "application/json")
        w.Write(productsJSON)
    case http.MethodPost:
        var newProduct Product
        bodyBytes, err := ioutil.ReadAll(r.Body)
        if err != nil {
            w.WriteHeader(http.StatusBadRequest)
            return
        }
        err = json.Unmarshal(bodyBytes, &newProduct)
        if err != nil {
            w.WriteHeader(http.StatusBadRequest)
            return
        }
        if newProduct.ProductID != 0 {
            w.WriteHeader(http.StatusBadRequest)
            return
        }
        newProduct.ProductID = getNextID()
        productList = append(productList, newProduct)
        w.WriteHeader(http.StatusCreated)
        return
    }
}


func main() {
    http.HandleFunc("/products", productsHandler)
    log.Fatal(http.ListenAndServe(":5000", nil))
}
```
