
# vim 配置文件

file location: ~/.vimrc

```
"this is comment

set number
set autoindent
set smartindent
set tabstop=4
set smarttab
set softtabstop=4
set shiftwidth=4
set expandtab
set hlsearch
set cursorline
set pastetoggle=<F3>

"for ctags
set tags=tags;
set autochdir

filetype on
filetype indent on
filetype plugin on
syntax on

autocmd FileType yaml,html setlocal ai ts=2 sw=2 et

"for NERDTree
"execute pathogen#infect()
"map <F10> :NERDTreeToggle<CR>

"auto close pair
:inoremap ( ()<ESC>i
:inoremap ) <c-r>=ClosePair(')')<CR>
":inoremap { {<CR>}<ESC>O
:inoremap { {}i
:inoremap } <c-r>=ClosePair('}')<CR>
:inoremap [ []<ESC>i
:inoremap ] <c-r>=ClosePair(']')<CR>
:inoremap " ""<ESC>i
:inoremap ' ''<ESC>i
function! ClosePair(char)
    if getline('.')[col('.') - 1] == a:char
		        return "\<Right>"
	else
			    return a:char
	endif
endfunction

:inoremap <TAB> <c-r>=SkipPair()<CR>
function SkipPair()
    if getline('.')[col('.') - 1] == ')' || getline('.')[col('.') - 1] == ']' || getline('.')[col('.') - 1] == '"' || getline('.')[col('.') - 1] == "'" || getline('.')[col('.') - 1] == '}'
        return "\<ESC>la"
    else
        return "\t"
    endif
endfunction

:inoremap <CR> <c-r>=BreakPair()<CR>
function BreakPair()
    ":let s:char = getline('.')[col('$')]
    if getline('.')[col('.') - 1] == '}'
        return "\<CR>\<ESC>O"
    else
        return "\<CR>"
endfunction
```
